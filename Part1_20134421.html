# LPS TO RAS Converter 
# we know that to convert PLS to RAS we should just multiply first two values of our list by -1(to change directions of the coordinates from right to left and from anterior to posterior) 
# https://www.slicer.org/wiki/Coordinate_systems
# to ease manipulation of list we convert it to np.array

def lps_to_ras(coordinate):
    #using comprehension list create temporary list with converted values 
    temp_list = [i*-1 if coordinate.index(i) != 2 else i for i in coordinate]
    return temp_list
    
# using the converter we can easily figure out the RAS coordinates 
lps_to_ras(coordinate_list)
